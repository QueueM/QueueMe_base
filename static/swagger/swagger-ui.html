package main

import (
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    client := &http.Client{}
    req, _ := http.NewRequest("GET", "https://api.queueme.net/api/v1/shops", nil)

    req.Header.Set("Authorization", "Bearer YOUR_API_KEY")
    req.Header.Set("Content-Type", "application/json")

    resp, err := client.Do(req)
    if err != nil {
        fmt.Println("Error:", err)
        return
    }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
Authentication
QueueMe API uses JWT (JSON Web Tokens) for authentication. Include your token in the Authorization header with the Bearer prefix.
Obtaining API Keys
Sign up at QueueMe Developer Portal to receive your API keys. You'll need to verify your account before receiving production keys.
Token Expiration
Access tokens expire after 24 hours. Use the refresh token endpoint to obtain a new access token.
Quick Start
  1. Register at the Developer Portal to get your API key
  2. Make sure to include your API key in the Authorization header
  3. Explore the endpoints below to understand the available operations
  4. Start with basic operations like listing shops or services
  5. Check the response codes and formats to ensure correct implementation
